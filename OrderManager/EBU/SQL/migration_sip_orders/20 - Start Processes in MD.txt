--- CHECK THAT ALL THE ELEMENTS HAVE BEEN ALREADY SENT TO GRANITE!!!

select c.alreadysenttogranite
  from stc_bundleorder_header h, stc_lineItem b, stc_lineitem c, stc_order_orchestration orc
 where h.ismigrated = 1
   and b.cworderid = h.cworderid
   and b.elementtypeinordertree = 'B'
   and b.provisioningFlag = 'PROVISIONING'
   and orc.cworderid = h.cworderid
   and h.ordernumber in (select parentordernumber from stc_om_home_sip)
   and c.cworderid = b.cworderid
   and c.elementtypeinordertree = 'C'
